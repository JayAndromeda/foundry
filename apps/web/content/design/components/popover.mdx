## Usage

```tsx
import {
  Popover,
  PopoverTrigger,
  PopoverTitle,
  PopoverContent,
} from "@foundry-ui/Popover";
```

```tsx
<Popover>
  <PopoverTrigger asChild>
    <Button variant="secondary">Open</Button>
  </PopoverTrigger>
  <PopoverContent>
    <PopoverTitle>Title</PopoverTitle>
    <div>The content for the popover.</div>
  </PopoverContent>
</Popover>
```
